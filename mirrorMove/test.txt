func (mr *MoveRepository) CreateMoveWithActions(move Dto.MoveCreate) error {
    newMove := models.Move{Name: move.Name}
    for _, actionID := range move.ActionIDs {
        var action models.Action
        if err := mr.db.First(&action, actionID).Error; err != nil {
            return err
        }
        newMove.Actions = append(newMove.Actions, action)
    }

    if err := mr.db.Create(&newMove).Error; err != nil {
        return err
    }

    return nil
}
